CREATE OR REPLACE TRIGGER PASSED_OR_NOT

AFTER INSERT OR UPDATE ON RESULT45
FOR EACH ROW

BEGIN
 
IF :OLD.MARKS IS NULL AND :NEW.MARKS < 50 THEN

INSERT INTO BACKPAPER45 VALUES (:NEW.ROLL,:NEW.SCODE);

END IF;

END IF;

END;
