DECLARE
EC EMP45.EMP_CODE%TYPE;
EN EMP45.EMP_NAME%TYPE;
BEGIN
EC := '&EC';
SELECT EMP_NAME INTO EN FROM EMP45 WHERE EMP_CODE = EC;
DBMS_OUTPUT.PUT_LINE('EMP NAME IS '||EN);
EXCEPTION
WHEN NO_DATA_FOUND THEN
DBMS_OUTPUT.PUT_LINE('NO EMPLOYEE FOUND');
END;
/



DECLARE
CURSOR EMP_CUR IS SELECT  * FROM EMP45;
I EMP_CUR%ROWTYPE;
EC EMP45.EMP_CODE%TYPE;
EN EMP45.EMP_NAME%TYPE;
BEGIN
EN := 'NOT FOUND';
EC := '&EC';
FOR I IN EMP_CUR
LOOP
IF I.EMP_CODE = EC THEN
EN := I.EMP_NAME;
END IF;
END LOOP;
IF EN = 'NOT FOUND' THEN
DBMS_OUTPUT.PUT_LINE('NO EMPLOYEE FOUND');
ELSE
DBMS_OUTPUT.PUT_LINE('EMP NAME IS ' || EN);
END IF;
END;
/


DECLARE
CURSOR EMP_CUR IS SELECT  * FROM EMP45;
I EMP_CUR%ROWTYPE;
EC EMP45.EMP_CODE%TYPE;
EN EMP45.EMP_NAME%TYPE;
CHK INT;
BEGIN
CHK := 0;
EC := '&EC';
EN := '&EN';
FOR I IN EMP_CUR
LOOP
IF I.EMP_CODE = EC THEN
CHK := 1;
END IF;
END LOOP;
IF CHK = 1 THEN
DBMS_OUTPUT.PUT_LINE(' EMPLOYEE ALREADY EXIST');
ELSE
OPEN EMP_CUR;
INSERT INTO EMP_CUR (EMP_CODE,EMP_NAME) VALUES(EC,EN);
CLOSE EMP_CUR;
END IF;
END;
/
